function addMessage(content, sender, isImage = false) { const msgDiv = document.getElementById('messages'); const div = document.createElement('div'); div.className = `message ${sender}`; if (sender === 'bestie') { div.innerHTML = `
🫂
${content}
`; } else { div.textContent = content; } if (isImage && content && (content.startsWith('http') || content.startsWith('data:'))) { div.innerHTML += `
`; } msgDiv.appendChild(div); msgDiv.scrollTop = msgDiv.scrollHeight; } function showTyping() { const msgDiv = document.getElementById('messages'); const typing = document.createElement('div'); typing.id = 'typing'; typing.className = 'typing'; typing.textContent = 'Bestie is thinking... 💭'; msgDiv.appendChild(typing); msgDiv.scrollTop = msgDiv.scrollHeight; return typing; } function wantsImage(text) { const lower = text.toLowerCase(); return /(draw|generate|make|create|imagine).*?(image|picture|photo|art|sticker|drawing)/.test(lower) || lower.includes("draw me") || lower.includes("image of"); } async function sendMessage() { const input = document.getElementById('userInput'); const promptText = input.value.trim(); if (!promptText) return; addMessage(promptText, 'user'); input.value = ''; const typing = showTyping(); try { const isImageRequest = wantsImage(promptText); const model = isImageRequest ? "gemini-2.5-flash-preview-image" : "gemini-2.5-flash"; const url = `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${API_KEY}`; const res = await fetch(url, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [{ parts: [{ text: promptText }] }] }) }); if (!res.ok) { const errorText = await res.text(); console.error("Gemini Error:", res.status, errorText); throw new Error("API request failed"); } const data = await res.json(); typing.remove(); let replyText = ""; let imageUrl = null; if (data.candidates && data.candidates[0] && data.candidates[0].content && data.candidates[0].content.parts) { for (let part of data.candidates[0].content.parts) { if (part.text) replyText += part.text + " "; if (part.inlineData) { imageUrl = `data:${part.inlineData.mimeType};base64,${part.inlineData.data}`; } } } if (imageUrl) { addMessage(imageUrl, 'bestie', true); setTimeout(() => { addMessage("Here you go, love! 🥰 How does this feel? Want me to change anything?", 'bestie'); }, 800); } else if (replyText.trim()) { addMessage(replyText.trim(), 'bestie'); } else { addMessage("I got a little confused there 💭 Can you try again?", 'bestie'); } } catch (err) { console.error("Full error:", err); typing.remove(); addMessage("Sorry sweetie, Bestie had a tiny connection glitch 💭 Try again in a moment!", 'bestie'); } } // Welcome message window.onload = () => { checkLogin(); const messagesDiv = document.getElementById('messages'); if (messagesDiv.children.length === 0) { addMessage("Hey love! 🫂 I'm your Bestie, now powered by Gemini. What's on your heart today? Poems, advice, or want me to draw something special for you? 💕", 'bestie'); } };
